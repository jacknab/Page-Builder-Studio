let currentStep = 1;
let formData = {};
let selectedLoc = 'cascade';

function openModal(loc){
  if(loc) selectedLoc = loc;
  currentStep = 1;
  formData = {};
  document.getElementById('modalOverlay').classList.add('open');
  document.body.style.overflow = 'hidden';
  renderStep();
}
function closeModal(){
  document.getElementById('modalOverlay').classList.remove('open');
  document.body.style.overflow = '';
}
function handleOverlayClick(e){
  if(e.target === document.getElementById('modalOverlay')) closeModal();
}

function stepsHTML(active){
  const labels = ['Info','Service','Confirm'];
  let s = '<div class="steps-indicator">';
  labels.forEach((l,i)=>{
    const n = i+1;
    let cls = n < active ? 'done' : n === active ? 'active' : '';
    s += `<div class="step-dot ${cls}">${n < active ? '✓' : n}</div>`;
    if(i < labels.length-1){
      s += `<div class="step-line ${n < active ? 'done':''}"></div>`;
    }
  });
  s += '</div>';
  return s;
}

function toggleSvc(id, el){
  formData.services = formData.services||[];
  const idx = formData.services.indexOf(id);
  if(idx > -1){ formData.services.splice(idx,1); el.classList.remove('selected'); }
  else { formData.services.push(id); el.classList.add('selected'); }
}
